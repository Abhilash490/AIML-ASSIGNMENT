import pandas as pd

# Load dataset (example: CSV file)
df = pd.read_csv("data.csv")

# Display top rows
print("\n--- Top 5 Rows ---")
print(df.head())

# Find column with highest value (numeric columns only)
numeric_cols = df.select_dtypes(include='number')
max_column = numeric_cols.max().idxmax()
max_value = numeric_cols.max().max()

print("\nColumn with Highest Value:", max_column)
print("Highest Value:", max_value)

# Count missing values
print("\n--- Missing Values Count ---")
print(df.isnull().sum())
