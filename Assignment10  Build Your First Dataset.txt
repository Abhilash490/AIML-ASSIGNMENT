import pandas as pd
import matplotlib.pyplot as plt

# Create dataset
data = {
    "Study Hours": [1,2,3,4,5,6,7,8],
    "Marks": [35,40,50,55,65,70,75,85]
}

df = pd.DataFrame(data)

print(df)

# Plot relationship
plt.scatter(df["Study Hours"], df["Marks"])
plt.xlabel("Study Hours")
plt.ylabel("Marks")
plt.title("Study Hours vs Marks")
plt.show()
